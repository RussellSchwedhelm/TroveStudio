# Supabase Setup for Trôve Studio

To get your database running, please execute the following SQL in your Supabase SQL Editor:

```sql
-- Create Products Table
CREATE TABLE products (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc'::text, NOW()) NOT NULL,
  title TEXT NOT NULL,
  price DECIMAL NOT NULL,
  image_url TEXT NOT NULL,
  category TEXT,
  stock_quantity INTEGER DEFAULT 0
);

-- Insert Initial Seed Data
INSERT INTO products (title, price, image_url, category)
VALUES 
  ('The Classic Ring', 2650, '/assets/images/trove_hero_3_1778846644095.png', 'Rings'),
  ('Chunky Hoops', 3850, '/assets/images/trove_hero_1_1778846613200.png', 'Earrings'),
  ('Statement Chain', 6400, '/assets/images/trove_hero_2_1778846628465.png', 'Necklaces');

-- Enable Row Level Security (RLS)
ALTER TABLE products ENABLE ROW LEVEL SECURITY;

-- Create a policy to allow public read access
CREATE POLICY "Allow public read access" ON products
  FOR SELECT USING (true);
```

-- Create News Table
CREATE TABLE news (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc'::text, NOW()) NOT NULL,
  title TEXT NOT NULL,
  date TEXT NOT NULL,
  category TEXT NOT NULL,
  excerpt TEXT NOT NULL,
  image_url TEXT NOT NULL
);

-- Create Journal Table
CREATE TABLE journals (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc'::text, NOW()) NOT NULL,
  title TEXT NOT NULL,
  subtitle TEXT,
  content TEXT,
  image_url TEXT NOT NULL,
  is_featured BOOLEAN DEFAULT false
);

-- Add Role to Profiles
ALTER TABLE profiles ADD COLUMN role TEXT DEFAULT 'user';

-- Enable RLS for news and journals
ALTER TABLE news ENABLE ROW LEVEL SECURITY;
ALTER TABLE journals ENABLE ROW LEVEL SECURITY;

-- Public read access for news and journals
CREATE POLICY "Allow public read access" ON news FOR SELECT USING (true);
CREATE POLICY "Allow public read access" ON journals FOR SELECT USING (true);

-- Admin full access for everything
CREATE POLICY "Admins have full access on products" ON products
  FOR ALL USING (
    EXISTS (SELECT 1 FROM profiles WHERE profiles.id = auth.uid() AND profiles.role = 'admin')
  );

CREATE POLICY "Admins have full access on news" ON news
  FOR ALL USING (
    EXISTS (SELECT 1 FROM profiles WHERE profiles.id = auth.uid() AND profiles.role = 'admin')
  );

CREATE POLICY "Admins have full access on journals" ON journals
  FOR ALL USING (
    EXISTS (SELECT 1 FROM profiles WHERE profiles.id = auth.uid() AND profiles.role = 'admin')
  );

### Environment Variables
Once the project is created, add these to your `.env.local` and Vercel settings:

- `NEXT_PUBLIC_SUPABASE_URL`: Your Supabase project URL.
- `NEXT_PUBLIC_SUPABASE_ANON_KEY`: Your Supabase anon key.
